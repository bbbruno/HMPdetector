69 void Creator::getFeatures(string name, int nTr, mat &totGravity, mat &totBody)
70 {
71  string fileName; // name of one trial of the modelling dataset
72 
73  for (int i = 0; i < nTr; i++)
74  {
75  // read all the modelling trials and concatenate the values along the axes
76  stringstream itos;
77  itos <<i+1;
78  fileName = datasetFolder + name + "/mod (" + itos.str() + ").txt";
79  cout<<"Open modelling trial: " <<fileName <<endl;
80  mat set;
81  mat gravity;
82  mat body;
83  ifstream trialFile(fileName.c_str());
84  for (string line; std::getline(trialFile, line); )
85  {
86  //DEBUG:cout<<"Line: " <<line <<endl;
87  mat actualSample;
88  actualSample = driver->extractActual(line);
89  createSet(actualSample, set);
90  }
91 
92  // reduce the noise on the sets by median filtering
93  int size = 3;
94  mat clean_set = set.t();
95  medianFilter(clean_set, size);
96  clean_set = clean_set.t();
97 
98  // separate gravity and body acc. by Chebyshev II low-pass filtering
99  mat tempgr = clean_set.t();
100  gravity = ChebyshevFilter(tempgr);
101  gravity = gravity.t();
102  body = clean_set - gravity;
103 
104  // create the datasets
105  mat time = createInterval(1, gravity.n_rows);
106  mat tmpGrav = join_rows(time, gravity);
107  mat tmpBody = join_rows(time, body);
108  totGravity = join_cols(totGravity, tmpGrav);
109  totBody = join_cols(totBody, tmpBody);
110  }
111 }

115 void Creator::generateModel(STmodel &motion)
116 {
117  mat totGravity;
118  mat totBody;
119  string GMMgravity;
120  string GMMbody;
121  string MuGr;
122  string MuBo;
123  string SGr;
124  string SBo;
125 
126  // create the gravity and body acc. datasets
127  cout<<endl <<"Creating the gravity and body acceleration datasets" <<endl;
128  getFeatures(motion.name, motion.nbModellingTrials, totGravity, totBody);
129 
130  // create the GMM+GMR model of the gravity component
131  {
132  cout<<endl <<"GMM+GMR model of the gravity component" <<endl;
133  GaussianMixture gg;
134  int nbVar = totGravity.n_cols;
135  int nbData = totGravity.n_rows;
136  nbData = (int) (nbData / motion.nbModellingTrials);
137  cout<<"Number of samples in the modelling trials: " <<nbData <<endl;
138 
139  // GMM phase
140  cout<<endl <<"GMM...";
141  gg.initEM_TimeSplitMat(motion.nbGravityGaussians, totGravity);
142  gg.doEM(totGravity);
143  GMMgravity = datasetFolder + motion.name + "GMMgravity.txt";
144  gg.saveParams(GMMgravity.c_str());
145  cout <<"done" <<endl;
146 
147  // GMR phase
148  cout<<endl <<"GMR...";
149  Vector inC(1), outC(nbVar - 1);
150  // input data for regression: time
151  inC(0) = 0;
152  // output data for regression: tri-axial acceleration
153  for (int i = 0; i < nbVar - 1; i++)
154  outC(i) = (float) (i + 1);
155  mat oneCol = createInterval(1,nbData);
156  Matrix *inData = new Matrix(oneCol);
157  Matrix *outSigma;
158  outSigma = new Matrix[nbData];
159  Matrix outData = gg.doRegression(*inData, outSigma, inC, outC);
160  MuGr = datasetFolder + motion.name + "MuGravity.txt";
161  SGr = datasetFolder + motion.name + "SigmaGravity.txt";
162  gg.saveRegressionResult(MuGr.c_str(),SGr.c_str(),*inData,outData,outSigma);
163  cout <<"done" <<endl;
164  }
165 
166  // create the GMM+GMR model of the body acc. component
167  {
168  cout<<endl <<"GMM+GMR model of the body acc. component" <<endl;
169  GaussianMixture gb;
170  int nbVar = totBody.n_cols;
171  int nbData = totBody.n_rows;
172  nbData = (int) (nbData / motion.nbModellingTrials);
173  cout<<"Number of samples in the modelling trials: " <<nbData <<endl;
174 
175  // GMM phase
176  cout<<endl <<"GMM...";
177  gb.initEM_TimeSplitMat(motion.nbBodyGaussians, totBody);
178  gb.doEM(totBody);
179  GMMbody = datasetFolder + motion.name + "GMMbody.txt";
180  gb.saveParams(GMMbody.c_str());
181  cout <<"done" <<endl;
182 
183  // GMR phase
184  cout<<endl <<"GMR...";
185  Vector inC(1), outC(nbVar - 1);
186  // input data for regression: time
187  inC(0) = 0;
188  // output data for regression: tri-axial acceleration
189  for (int i = 0; i < nbVar - 1; i++)
190  outC(i) = (float) (i + 1);
191  mat oneCol = createInterval(1,nbData);
192  Matrix *inData = new Matrix(oneCol);
193  Matrix *outSigma;
194  outSigma = new Matrix[nbData];
195  Matrix outData = gb.doRegression(*inData, outSigma, inC, outC);
196  MuBo = datasetFolder + motion.name + "MuBody.txt";
197  SBo = datasetFolder + motion.name + "SigmaBody.txt";
198  gb.saveRegressionResult(MuBo.c_str(),SBo.c_str(),*inData,outData,outSigma);
199  cout <<"done" <<endl;
200  }
201 }
