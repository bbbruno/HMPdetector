102  window = matrix.submat(r, i - step, r, i + step);
103  else
104  {

106  window.submat(0,step,0,size-1) = matrix.submat(r,0,r,step);
107  else
108  if (i > (matrix.n_cols - 1 - step))
109  window.submat(0, 0, 0, step) =
110  matrix.submat(r, i - step, r, matrix.n_cols - 1);
