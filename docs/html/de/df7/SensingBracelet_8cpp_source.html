211 void SensingBracelet::updateInterval(int& i,int& nS,float &p,float &pp,int f,string rF)
212 {
213  // 1) active
214  if(p > 0)
215  {
216  // 1.1) new active: create a new interval
217  if(setI[i].open == false)
218  {
219  setI[i].startInterval(p, nS, pp);
220  cout<<setI[i].HMPname <<": open interval" <<endl;
221  }
222  // 1.2.1) already active and rising
223  else if(p >= setI[i].possibility)
224  {
225  setI[i].end[0] = nS;
226  setI[i].possibility = p;
227  setI[i].risingTime++;
228  //DEBUG: cout<<"Rising" <<endl;
229  }
230  // 1.2.2) already active and descending
231  else
232  {
233  setI[i].risingTime--;
234  // if the descent is not smooth, the interval is NOT valid
235  if(p > pp)
236  {
237  setI[i].open = false;
238  cout<<setI[i].HMPname <<": false positive - NO bell shape" <<endl;
239  }
240  // check the simmetry of the possibility curve
241  if((setI[i].risingTime == 0) && (p < setI[i].refPossibility))
242  {
243  setI[i].open = false;
244  cout<<setI[i].HMPname <<": false positive - NO symmetry" <<endl;
245  }
246  // if the above constraints are satisfied, publish the result on PEIS
247  if((setI[i].risingTime >= 0) && (p == setI[i].refPossibility))
248  {
249  // define the ending time of the interval
250  setI[i].setEnd(setI[i].end[0]);
251 
252  // publish the interval info
253  if (f == 0)
254  // on file
255  reportSensingBracelet(i, rF);
256  if (f == 1)
257  // on PEIS
258  publishSensingBracelet(i);
259 
260  // close the interval
261  setI[i].open = false;
262  }
263  }
264  }
265  // 2) inactive
266  // (if the interval should be closed: open == true && refPossibility == 0)
267  else if((setI[i].open == true) && (setI[i].refPossibility == 0))
268  {
269  // define the ending time of the interval
270  setI[i].setEnd(setI[i].end[0]);
271 
272  // publish the interval info
273  if (f == 0)
274  // on file
275  reportSensingBracelet(i, rF);
276  if (f == 1)
277  // on PEIS
278  publishSensingBracelet(i);
279 
280  // close the interval
281  setI[i].open = false;
282  }
283 }

342 void SensingBracelet::offlineSensingBracelet(string path, string testFile)
343 {
344  vector<float> possibilities; // models possibilities
345  vector<float> past_possibilities; // models previous possibilities
346  int nSamples = 0; // number of acquired samples
347 
348  // create result file (same folder of the possibilities file)
349  string rF = path + "Rres_" + testFile;
350 
351  // initialize the possibilities and PREVIOUSpossibilities
352  for(int i = 0; i < nbM; i++)
353  {
354  possibilities.push_back(0);
355  past_possibilities.push_back(0);
356  }
357 
358  // read & analyze recorded possibilities one line at the time
359  string pFile = path + testFile;
360  ifstream pf(pFile.c_str());
361  cout <<"Reading possibilities in: " <<pFile <<endl;
362  while (!pf.eof())
363  {
364  // update the values of the past_possibilities
365  for(int i = 0; i < nbM; i++)
366  past_possibilities[i] = possibilities[i];
367 
368  // read the new possibilities
369  for(int i = 0; i < nbM; i++)
370  {
371  pf>>possibilities[i];
372  //DEBUG: cout<<possibilities[i] <<" ";
373  }
374  //DEBUG: cout<<endl;
375  if (!pf)
376  break;
377 
378  // extract/update the intervals of activation for each activity
379  for(int i = 0; i < nbM; i++)
380  updateInterval(i,nSamples,possibilities[i],past_possibilities[i],0,rF);
381 
382  nSamples = nSamples + 1;
383  }
384  pf.close();
385 
386  // analyze intervals that are still open at the end of the file
387  for (int i = 0; i < nbM; i++)
388  {
389  if((setI[i].open == true) && (setI[i].refPossibility == 0))
390  {
391  // define the ending time of the interval
392  setI[i].setEnd(setI[i].end[0]);
393 
394  // publish the interval info on file
395  reportSensingBracelet(i, rF);
396 
397  // close the interval
398  setI[i].open = false;
399  }
400  }

406 void SensingBracelet::onlineSensingBracelet(char* port)
407 {
408  int nSamples = 0; // number of acquired samples
409  string sample; // current sample acquired via USB
410  mat actsample; // current sample in matrix format
411  int ax, ay, az; // accelerometer current sample components
412  int gx, gy, gz; // gyroscope current sample components
413  char dev; // flag --> device type
414  string motion; // flag --> level of motion at the wrist
415  vector<float> poss; // models possibilities
416  vector<float> past_poss; // models previous possibilities
417 
418  string waste = " ";
419 
420  // instantiate and initialize a Classifier
421  string dF = datasetFolder.substr(0,datasetFolder.length()-1);
422  dF = dF.substr(9);
423  Classifier hC(dF);
424  mat window = zeros<mat>(hC.window_size, 3);
425  mat gravity = zeros<mat>(hC.window_size, 3);
426  mat body = zeros<mat>(hC.window_size, 3);
427 
428  // initialize the possibilities and past possibilities
429  for(int i = 0; i < nbM; i++)
430  {
431  poss.push_back(0);
432  past_poss.push_back(0);
433  }
434 
435  // set up the serial communication (read-only)
436  SerialOptions options;
437  options.setDevice(port);
438  options.setBaudrate(9600);
439  options.setTimeout(seconds(1));
440  options.setParity(SerialOptions::noparity);
441  options.setCsize(8);
442  options.setFlowControl(SerialOptions::noflow);
443  options.setStopBits(SerialOptions::one);
444  SerialStream serial(options);
445  serial.exceptions(ios::badbit | ios::failbit);
446 
447  // extract known activities intervals from the stream of raw acceleration data
448  while(1)
449  {
450  try
451  {
452  // read the current sample
453  getline(serial,sample);
454  istringstream stream(sample);
455  stream >>dev >>ax >> ay >>az >>gx >>gy >>gz >>motion;
456  actsample <<ax <<ay <<az;
457  //DEBUG: cout<<"Acquired: " <<ax <<" " <<ay <<" " <<az <<" ";
458 
459  // update the window of samples to be analyzed
460  hC.createWindow(actsample, window, hC.window_size, nSamples);
461  if (nSamples >= hC.window_size)
462  {
463  // analyze the window and compute the models possibilities
464  for(int i = 0; i < nbM; i++)
465  past_poss[i] = poss[i];
466  hC.analyzeWindow(window, gravity, body);
467  hC.compareAll(gravity, body, poss);
468 
469  // publish the dynamic tuples
470  hC.publishDynamic(poss);
471 
472  /**************************************************************
473  // ACCURATE A-POSTERIORI ACTIVITY ANALYSIS
474  // extract/update the intervals of activation for each activity
475  for(int i = 0; i < nbM; i++)
476  updateInterval(i,nSamples,poss[i],past_poss[i],1,waste);
477  **************************************************************/
478 
479  // QUICK AND DIRTY ANALYSIS
480  // extract/update the intervals of activation for each activity
481  for(int i = 0; i < nbM; i++)
482  simpleInterval(i,nSamples,poss[i],past_poss[i]);
483  }
484  }
485  catch(TimeoutException&)
486  {
487  serial.clear();
488  cerr<<"Timeout occurred"<<endl;
489  }
490  }
491 }
