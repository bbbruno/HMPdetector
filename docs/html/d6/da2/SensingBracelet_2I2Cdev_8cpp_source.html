140 int8_t I2Cdev::readBits(uint8_t devAddr, uint8_t regAddr, uint8_t bitStart, uint8_t length, uint8_t *data, uint16_t timeout) {
141  // 01101001 read byte
142  // 76543210 bit numbers
143  // xxx args: bitStart=4, length=3
144  // 010 masked
145  // -> 010 shifted
146  uint8_t count, b;
147  if ((count = readByte(devAddr, regAddr, &b, timeout)) != 0) {
148  uint8_t mask = ((1 << length) - 1) << (bitStart - length + 1);
149  b &= mask;
150  b >>= (bitStart - length + 1);
151  *data = b;
152  }

535 bool I2Cdev::writeBits(uint8_t devAddr, uint8_t regAddr, uint8_t bitStart, uint8_t length, uint8_t data) {
536  // 010 value to write
537  // 76543210 bit numbers
538  // xxx args: bitStart=4, length=3
539  // 00011100 mask byte
540  // 10101111 original value (sample)
541  // 10100011 original & ~mask
542  // 10101011 masked | value
543  uint8_t b;
544  if (readByte(devAddr, regAddr, &b) != 0) {
545  uint8_t mask = ((1 << length) - 1) << (bitStart - length + 1);
546  data <<= (bitStart - length + 1); // shift data into correct position
547  data &= mask; // zero all non-important bits in data
548  b &= ~(mask); // zero all important bits in existing byte
549  b |= data; // combine data with existing byte
550  return writeByte(devAddr, regAddr, b);
551  } else {
552  return false;
553  }
554 }

703  #elif (I2CDEV_IMPLEMENTATION == I2CDEV_ARDUINO_WIRE && ARDUINO >= 100)
704  status = Wire.endTransmission();
705  #endif
706  #ifdef I2CDEV_SERIAL_DEBUG

731  void Fastwire::setup(int khz, boolean pullup) {
732  TWCR = 0;
733  #if defined(__AVR_ATmega168__) || defined(__AVR_ATmega8__) || defined(__AVR_ATmega328P__)
734  // activate internal pull-ups for twi (PORTC bits 4 & 5)
735  // as per note from atmega8 manual pg167
736  if (pullup) PORTC |= ((1 << 4) | (1 << 5));
737  else PORTC &= ~((1 << 4) | (1 << 5));
738  #elif defined(__AVR_ATmega644P__) || defined(__AVR_ATmega644__)
739  // activate internal pull-ups for twi (PORTC bits 0 & 1)
740  if (pullup) PORTC |= ((1 << 0) | (1 << 1));
741  else PORTC &= ~((1 << 0) | (1 << 1));
742  #else
743  // activate internal pull-ups for twi (PORTD bits 0 & 1)
744  // as per note from atmega128 manual pg204
745  if (pullup) PORTD |= ((1 << 0) | (1 << 1));
746  else PORTD &= ~((1 << 0) | (1 << 1));
747  #endif
748 
749  TWSR = 0; // no prescaler => prescaler = 1
750  TWBR = ((16000L / khz) - 16) / 2; // change the I2C clock rate
751  TWCR = 1 << TWEN; // enable twi module, no interrupt
752  }

754  byte Fastwire::write(byte device, byte address, byte value) {
755  byte twst, retry;
756 
757  retry = 2;
758  do {
759  TWCR = (1 << TWINT) | (1 << TWEN) | (1 << TWSTO) | (1 << TWSTA);
760  if (!waitInt()) return 1;
761  twst = TWSR & 0xF8;
762  if (twst != TW_START && twst != TW_REP_START) return 2;
763 
764  TWDR = device & 0xFE; // send device address without read bit (1)
765  TWCR = (1 << TWINT) | (1 << TWEN);
766  if (!waitInt()) return 3;
767  twst = TWSR & 0xF8;
768  } while (twst == TW_MT_SLA_NACK && retry-- > 0);
769  if (twst != TW_MT_SLA_ACK) return 4;
770 
771  TWDR = address; // send data to the previously addressed device
772  TWCR = (1 << TWINT) | (1 << TWEN);
773  if (!waitInt()) return 5;
774  twst = TWSR & 0xF8;
775  if (twst != TW_MT_DATA_ACK) return 6;
776 
777  TWDR = value; // send data to the previously addressed device
778  TWCR = (1 << TWINT) | (1 << TWEN);
779  if (!waitInt()) return 7;
780  twst = TWSR & 0xF8;
781  if (twst != TW_MT_DATA_ACK) return 8;
782 
783  return 0;
